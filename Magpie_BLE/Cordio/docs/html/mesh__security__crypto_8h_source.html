320 static inline void meshSecBuildNonce(uint8_t nonceType, uint8_t ttlCtlAszMicPad, uint16_t src,
321  uint16_t dstPad, meshSeqNumber_t seqNumber, uint32_t ivIndex,
322  uint8_t *pNonceBuff)
323 {
324  /* Set nonce type. */
325  *(pNonceBuff++) = nonceType;
326 
327  if (nonceType == MESH_SEC_NONCE_PROXY)
328  {
329  /* CTL-TTL byte is padded with 0x00. */
330  *(pNonceBuff++) = 0x00;
331  }
332  else
333  {
334  /* Set CTL-TTL byte or ASZMIC_PAD. */
335  *(pNonceBuff++) = ttlCtlAszMicPad;
336  }
337 
338  /* Set sequence number. */
339  *(pNonceBuff++) = (uint8_t)(seqNumber >> 16);
340  *(pNonceBuff++) = (uint8_t)(seqNumber >> 8);
341  *(pNonceBuff++) = (uint8_t)(seqNumber);
342 
343  /* Set source address. */
344  *(pNonceBuff++) = (uint8_t)(src >> 8);
345  *(pNonceBuff++) = (uint8_t)(src);
346 
347  /* Set destination address or PAD. */
348  *(pNonceBuff++) = (uint8_t)(dstPad >> 8);
349  *(pNonceBuff++) = (uint8_t)(dstPad);
350 
351  /* Set IV index. */
352  *(pNonceBuff++) = (uint8_t)(ivIndex >> 24);
353  *(pNonceBuff++) = (uint8_t)(ivIndex >> 16);
354  *(pNonceBuff++) = (uint8_t)(ivIndex >> 8);
355  *(pNonceBuff++) = (uint8_t)(ivIndex);
356 }
